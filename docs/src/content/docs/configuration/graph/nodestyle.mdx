Here, you can find all configuration options related to the styling of nodes in the site graph.

## Basic Node

This is the styling of a basic node, with no additional styles applied to it.
Familiarize yourself with its shape, color, and size, before you move on to more advanced styles.

<GraphExample config={{}} baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"node/"} />


## Shape Config
You can change the shape of the node by setting the `shape` property.
When using the `polygon` or `star` shape, you can also set the number of points using the `shapePoints` property;
this property is ignored for other shapes.

<GraphExample id="shapeSelect" config={{ nodeDefaultStyle: { shape: "square", shapePoints: 0 } }} baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"node/"}
	buttons={{
		"Circle": { config: { nodeDefaultStyle: { shape: 'circle', shapePoints: 'undefined' } } },
		"Square": { config: { nodeDefaultStyle: { shape: 'square', shapePoints: 'undefined' } } },
		"Triangle": { config: { nodeDefaultStyle: { shape: 'triangle', shapePoints: 'undefined' } } },
		"Polygon": { config: { nodeDefaultStyle: { shape: 'polygon', shapePoints: 5 } } },
		"Star": { config: { nodeDefaultStyle: { shape: 'star', shapePoints: 5 } } },
	}}
	sliders={{
		"nodeDefaultStyle.shapePoints": {
			"title": "shapePoints",
			"min": 3,
			"max": 10,
			"step": 1,
			"val": 5,
		}
	}}
/>

You can change the size of the node by setting the `shapeSize` property.

## Shape Rotation
You can change the rotation of the node by setting the `shapeRotation` property.
This property can either receive a number input (in degrees), or `random` to randomize the rotation of the node.

<GraphExample id="rotationSelect" config={{ nodeDefaultStyle: { shapeRotation: 72, shape: "star", shapePoints: 5 } }} baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"node/"}
	sliders={{
		"nodeDefaultStyle.shapeRotation": {
			"title": "shapeRotation",
			"min": 0,
			"max": 360,
			"step": 1,
			"val": 72,
		}
	}}
/>


## Corner Radius
You can change the corner radius of the node by setting the `cornerRadius` property,
which works similarly to the CSS `border-radius` property.
This property is only applicable to polygon or star derived shapes (e.g. `square`, `triangle`, `polygon`, `star`).
Input can either be a number (in pixels), or a string with a percentage value (e.g. "5.3%").

<GraphExample id="shapeCornerRadius"
			  config={{
				  nodeDefaultStyle: {
					  shape: "square",
					  shapeCornerRadius: "4%",
					  cornerType: "round"
				  }
			  }}
			  baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"node/"}
			  buttons={{
				  "Circle": { config: { nodeDefaultStyle: { shape: 'circle', shapePoints: 'undefined' } } },
				  "Square": { config: { nodeDefaultStyle: { shape: 'square', shapePoints: 'undefined' } } },
				  "Triangle": { config: { nodeDefaultStyle: { shape: 'triangle', shapePoints: 'undefined' } } },
				  "Polygon": { config: { nodeDefaultStyle: { shape: 'polygon', shapePoints: 5 } } },
				  "Star": { config: { nodeDefaultStyle: { shape: 'star', shapePoints: 5 } } },
				  "Regular Corners": { class: "break-line corner-type-button", config: { nodeDefaultStyle: { cornerType: 'normal' } } },
				  "Rounded Corners": { class: "corner-type-button", config: { nodeDefaultStyle: { cornerType: 'round' } } },
				  "Beveled Corners": { class: "corner-type-button", config: { nodeDefaultStyle: { cornerType: 'bevel' } } },
			  }}
			  sliders={{
				  "nodeDefaultStyle.shapeCornerRadius": {
					  "title": "shapeCornerRadius",
					  "min": 0,
					  "max": 100,
					  "step": 0.1,
					  "val": 4,
					  "type": "percentage"
				  }
			  }}
/>

## Node Scaling

You can change the node scaling by setting the `scale` property.

<GraphExample id="nodeScale"
			  config={{
				  scale: 1,
				  nodeDefaultStyle: {
					  shapeSize: 10,
					  nodeScale: 1
				  }
			  }}
			  baseConfig={{ ...focusSingleNode, repelForce: 5 }} sitemap={twoNodeSitemap} slug={"node/"}
			  sliders={{
				  "nodeDefaultStyle.nodeScale": {
					  "title": "nodeScale",
					  "min": 0.1,
					  "max": 5,
					  "step": 0.1,
					  "val": 1
				  }
			  }}
/>


The node can also be scaled based on the amount of neighbors it has, this is configurable via
the `neighborScale` property.

<GraphExample id="neighborScale"
			  config={{
				  scale: 1,
				  nodeDefaultStyle: {
					  neighborScale: 1
				  }
			  }}
			  baseConfig={focusSingleNode} sitemap={starNodeSitemap} slug={"node/"}
			  sliders={{
				  "nodeDefaultStyle.neighborScale": {
					  "title": "neighborScale",
					  "min": 0,
					  "max": 50,
					  "step": 0.1,
					  "val": 1
				  }
			  }}
/>
