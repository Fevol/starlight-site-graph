Here, you can find all configuration options related to the styling of nodes in the site graph.

## Basic Node

This is the styling of a basic node, with no additional styles applied to it.
Familiarize yourself with its shape, color, and size, before you move on to more advanced styles.

<GraphExample config={{}} baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"/node/"} />


## Shape Config
The shape of the node can be set with the `shape` property.
When using the `polygon` or `star` shape, it is also possible to set the number of points added to the shape
using the `shapePoints` property; this property is ignored for other shapes.

<GraphExample id="shapeSelect"
  	config={{ graphConfig: { nodeDefaultStyle: { shape: "square", shapePoints: 0 } }}}
	baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"/node/"}
	buttons={{
		"Circle": { config: { graphConfig: { nodeDefaultStyle: { shape: 'circle', shapePoints: 'undefined' } } } },
		"Square": { config: { graphConfig: { nodeDefaultStyle: { shape: 'square', shapePoints: 'undefined' } } } },
		"Triangle": { config: { graphConfig: { nodeDefaultStyle: { shape: 'triangle', shapePoints: 'undefined' } } } },
		"Polygon": { config: { graphConfig: { nodeDefaultStyle: { shape: 'polygon', shapePoints: 5 } } } },
		"Star": { config: { graphConfig: { nodeDefaultStyle: { shape: 'star', shapePoints: 5 } } } },
	}}
	sliders={{
		"graphConfig.nodeDefaultStyle.shapePoints": {
			"title": "shapePoints",
			"min": 3,
			"max": 10,
			"step": 1,
			"val": 5,
		}
	}}
/>

## Size Config
The (absolute) size of the node can be set with the `shapeSize` property.
This size is further affected by the [`nodeScale`](#node-scaling) and [`neighborScale`](#neighbor-scaling).

<GraphExample id="sizeSelect"
  	config={{ graphConfig: { nodeDefaultStyle: { shapeSize: 10 } }}}
  	baseConfig={overrideConfig(focusSingleNode, { graphConfig: { scale: 1, repelForce: 5 }})} sitemap={twoNodeSitemap} slug={"/node/"}
	sliders={{
		"graphConfig.nodeDefaultStyle.shapeSize": {
			"title": "shapeSize",
			"min": 1,
			"max": 50,
			"step": 1,
			"val": 10,
		}
	}}
/>


## Color Config
With the `shapeColor` property, you can set the color of the node.
This property uses one of the predefined color types, which can be found below.

:::note[Note]
Changing the actual colors used for each color type can be done
by [changing the CSS variables](/starlight-site-graph/configuration/css/variables#nodes).
:::

<GraphExample id="colorSelect"
	config={{ graphConfig: { nodeDefaultStyle: { shapeColor: "nodeColor" } }}}
	baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"/node/"}
	dropdowns={{
		"shapeColor": nodeColorsDropdown(["nodeDefaultStyle", "nodeCurrentStyle"], "shapeColor", graphColors),
	}}
    dropdownDefaults={{
		"shapeColor": 0
	}}
/>


## Shape Rotation
The rotation of a node can be set with the `shapeRotation` property.
This property accepts either a number input (given as degrees), or `random` to randomize the rotation of the node.

<GraphExample id="rotationSelect"
	config={{ graphConfig: { nodeDefaultStyle: { shapeRotation: 72, shape: "star", shapePoints: 5 } }}}
    baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"/node/"}
	sliders={{
		"graphConfig.nodeDefaultStyle.shapeRotation": {
			"title": "shapeRotation",
			"min": 0,
			"max": 360,
			"step": 1,
			"val": 72,
		}
	}}
/>


## Corner Radius
With the `cornerRadius` property, you can set the corner radius of the node,
which works similarly to the CSS `border-radius` property.
This property is only applicable to polygon or star-derived shapes (e.g. `square`, `triangle`, `polygon`, `star`),
other shapes (e.g. `circle`) are not affected by this property.
Input can either be a number (in pixels), or a percentage value represented as a string (e.g. "5.3%").

<GraphExample id="shapeCornerRadius"
			  config={{
				graphConfig: {
					nodeDefaultStyle: {
						shape: "square",
						shapeCornerRadius: "4%",
						cornerType: "round"
					}
				}
			  }}
			  baseConfig={focusSingleNode} sitemap={singleNodeSitemap} slug={"/node/"}
			  buttons={{
				  "Circle": { config: { graphConfig: { nodeDefaultStyle: { shape: 'circle', shapePoints: 'undefined' } } } },
				  "Square": { config: { graphConfig: { nodeDefaultStyle: { shape: 'square', shapePoints: 'undefined' } } } },
				  "Triangle": { config: { graphConfig: { nodeDefaultStyle: { shape: 'triangle', shapePoints: 'undefined' } } } },
				  "Polygon": { config: { graphConfig: { nodeDefaultStyle: { shape: 'polygon', shapePoints: 5 } } } },
				  "Star": { config: { graphConfig: { nodeDefaultStyle: { shape: 'star', shapePoints: 5 } } } },
				  "Regular Corners": { class: "break-line corner-type-button", config: { graphConfig: { nodeDefaultStyle: { cornerType: 'normal' } } } },
				  "Rounded Corners": { class: "corner-type-button", config: { graphConfig: { nodeDefaultStyle: { cornerType: 'round' } } } },
				  "Beveled Corners": { class: "corner-type-button", config: { graphConfig: { nodeDefaultStyle: { cornerType: 'bevel' } } } },
			  }}
			  sliders={{
				  "graphConfig.nodeDefaultStyle.shapeCornerRadius": {
					  "title": "shapeCornerRadius",
					  "min": 0,
					  "max": 100,
					  "step": 0.1,
					  "val": 4,
					  "type": "percentage"
				  }
			  }}
/>


## Shape Stroke
Similarly to the `shape` properties, you can also set the size, color and corner radius of the stroke
using the `strokeWidth`, `strokeColor`, and `strokeRadius` properties.

<GraphExample id="shapeStroke"
			  config={{
				graphConfig: {
					nodeDefaultStyle: {
						shape: "square",
						shapeColor: "nodeColor1",
						strokeWidth: 2,
						strokeColor: "inherit",
						strokeCornerRadius: "4%",
						cornerType: "round"
					},
				}
			  }}
			  baseConfig={overrideConfig(focusSingleNode, { graphConfig: { nodeCurrentStyle: { shapeColor: "nodeColor1" } }})}

			  }}
			  dropdownDefaults={{

			  }}
			  sliders={{
				  "graphConfig.nodeDefaultStyle.strokeWidth": {
					  "title": "strokeWidth",
					  "min": 0,
					  "max": 20,
					  "step": 1,
					  "val": 2
				  },
				  "graphConfig.nodeDefaultStyle.strokeCornerRadius": {
					  "title": "strokeCornerRadius",
					  "min": 0,
					  "max": 100,
					  "step": 0.1,
					  "val": 4,
					  "type": "percentage"

			  }}
			  buttons={{

## Collider Scaling

The collider of a node is an invisible area around the node, which prevents other nodes from overlapping with it.
By default, the collider is identical to the size of the node itself, and only the simulation forces are responsible
for preventing nodes from overlapping.

However, if you'd like to have a larger area around the node that prevents other nodes from overlapping with it,
you can increase the size of the collider using the `colliderScale` property.

<GraphExample id="colliderScale"
			  config={{
				graphConfig: {
					repelForce: 0.1,
					nodeDefaultStyle: {
						colliderScale: 1
					}
				}
			  }}
			  baseConfig={overrideConfig(focusSingleNode, { graphConfig: { scale: 0.7, depth: 5 }})} sitemap={fourtyNodeSitemap} slug={"/node/"}
			  sliders={{
				  "graphConfig.nodeDefaultStyle.colliderScale": {
					  "title": "colliderScale",
					  "min": 0,
					  "max": 15,
					  "step": 1,
					  "val": 1
				  }
			  }}
/>


## Node Scaling

Using the `nodeScale` property, you can change how much a node is scaled with respect to the `shapeSize` property.
This can be useful if you want to make a specific category of nodes larger or smaller in comparison to other nodes
(e.g. making the _current page's_ node larger than the other nodes with `{ nodeCurrentStyle: { nodeScale: 2 } }`).

<GraphExample id="nodeScale"
			  config={{
				graphConfig: {
					scale: 1,
					nodeDefaultStyle: {
						shapeSize: 10,
						nodeScale: 1
					}
				}
			  }}
			  baseConfig={overrideConfig(focusSingleNode, { graphConfig: { repelForce: 5, depth: 5 }})} sitemap={twoNodeSitemap} slug={"/node/"}
			  sliders={{
				  "graphConfig.nodeDefaultStyle.nodeScale": {
					  "title": "nodeScale",
					  "min": 0.1,
					  "max": 5,
					  "step": 0.1,
					  "val": 1
				  }
			  }}
/>

## Neighbor Scaling

A node can have different amount of links compared to other nodes.
If you want a node to be larger than its surrounding nodes when it has more links,
you can do so by increasing the `neighborScale` property.

Internally, this property multiplies the `nodeScale` by a smoothed exponential function of the number of links the node has.

<GraphExample id="neighborScale"
			  config={{
				graphConfig: {
					scale: 1,
					nodeDefaultStyle: {
						neighborScale: 1
					}
				}
			  }}
			  baseConfig={focusSingleNode} sitemap={starNodeSitemap} slug={"/node/"}
			  sliders={{
				  "graphConfig.nodeDefaultStyle.neighborScale": {
					  "title": "neighborScale",
					  "min": 0,
					  "max": 50,
					  "step": 0.1,
					  "val": 1
				  }
			  }}
/>
